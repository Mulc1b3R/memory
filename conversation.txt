Use code with caution.The Python Scriptpythonimport matplotlib.pyplot as plt
import librosa
import librosa.display
import numpy as np

# 1. Load your audio file (Replace 'your_track.mp3' with your actual file name)
# 'sr=None' preserves the original sample rate of your audio
audio_path = 'your_track.mp3'
y, sr = librosa.load(audio_path, sr=None)

# 2. Convert the audio into frequency data over time (Short-Time Fourier Transform)
# n_fft controls frequency resolution; hop_length controls time resolution
stft_matrix = librosa.stft(y, n_fft=2048, hop_length=512)
stft_db = librosa.amplitude_to_db(np.abs(stft_matrix), ref=np.max)

# 3. Set up the plot layout
plt.figure(figsize=(12, 6))

# 4. Display the spectrogram
# x_axis='time' maps the song duration, y_axis='log' groups frequencies logically for human ears
img = librosa.display.specshow(stft_db, sr=sr, x_axis='time', y_axis='log', cmap='magma')

# 5. Add visual polish
plt.colorbar(img, format="%+2.0f dB")
plt.title('Track Frequency Energy Over Time (Spectrogram)', fontsize=14, pad=15)
plt.xlabel('Time (Seconds)', fontsize=12)
plt.ylabel('Frequency (Hz)', fontsize=12)

# Adjust frequency axis limits to focus on the human hearing range (20 Hz - 20 kHz)
plt.ylim(20, 20000)

plt.tight_layout()
plt.show()

def studio_filter(audio_array, cutoff_freq, filter_type='low', sr=48000):
    """Clean Butterworth filter for surgical frequency control"""
    nyquist = 0.5 * sr
    normal_cutoff = cutoff_freq / nyquist
    sos = butter(6, normal_cutoff, btype=filter_type, output='sos')
    return sosfiltfilt(sos, audio_array, axis=0)

def valve_saturation(audio_array, drive_db, blend=0.8):
    """Hyperbolic tangent waveshaper to synthesize tube amp overtones"""
    gain = 10 ** (drive_db / 20.0)
    driven = audio_array * gain
    saturated = np.tanh(driven)
    return (blend * saturated) + ((1.0 - blend) * audio_array)

def tape_sheen(audio_array, intensity=1.5):
    """Asymmetric exponential wave squasher for tape compression sheen"""
    return np.where(audio_array > 0, 1.0 - np.exp(-audio_array * intensity), -(1.0 - np.exp(audio_array * intensity)))

def align_track(ref_audio, tgt_audio):
    """Sample-accurate cross-correlation phase alignment"""
    correlation = correlate(ref_audio, tgt_audio, mode='full')
    lag = np.argmax(correlation) - (len(tgt_audio) - 1)
    if lag > 0:
        return np.pad(tgt_audio, (lag, 0))[:-lag]
    elif lag < 0:
        return np.append(tgt_audio[-lag:], np.zeros(-lag))
    return tgt_audio

def generate_stochastic_mutation_acid(output_path, total_blocks=4, bpm=120, base_freq=55.0):
    """
    Generates a specified number of 8-bar blocks where every single audio parameter
    mutates algorithmically and stochastically on execution, then stitches them.
    """
    sr = 48000
    beats_per_bar = 4
    beats_per_second = bpm / 60.0
    
    # 120 BPM Math: 1 bar = 2.0s | 8 bars = 16.0s per structural block
    seconds_per_bar = beats_per_bar / beats_per_second 
    block_duration = 8.0 * seconds_per_bar  
    samples_per_block = int(sr * block_duration)
    sixteenth_note_duration = 1.0 / (beats_per_second * 4) # 0.125s
    
    master_audio_array = np.array([], dtype=np.float32)
    print(f"\n=== WAKING THE STOCHASTIC MUTATION ENGINE ===")
    print(f"Generating {total_blocks} Blocks ({total_blocks * 8} Bars) | Grid: {bpm} BPM")

    for block_num in range(1, total_blocks + 1):
        # 1. Stochastic Parameter Selection (The Script Mutation)
        osc_type = random.choice(["sawtooth", "square", "triangle"])
        lfo_speed = random.choice([0.125, 0.25, 0.50, 1.00, 2.00]) # From ambient drift to hyper-scream
        input_gain = random.uniform(3.5, 7.0) # Randomized transistor saturation drive
        Q_resonance = random.uniform(7.5, 11.0) # Random sharpness of the circuit peak
        
        # Stochastic Pitch Matrix Generator (Alters the Fatboy Slim grid on the fly)
        # Selects a random combination of roots, octaves, minor 3rds, 4ths, and flat 5ths
        intervals_pool = [1.0, 2.0, 1.2, 1.33, 1.41, 1.5, 0.5]

        t = np.arange(samples_per_block) / sr
        phase = np.zeros(samples_per_block)
        current_phase = 0.0
        
        # 2. Render Algorithmic Pitch Path
        for i in range(samples_per_block):
            current_time = i / sr
            step_index = int(current_time / sixteenth_note_duration)
            pattern_note = pitch_intervals[step_index % len(pitch_intervals)]

            current_phase += 2.0 * np.pi * current_freq / sr
            phase[i] = current_phase

            raw_oscillator = signal.sawtooth(phase)
        elif osc_type == "square":

            raw_oscillator = signal.sawtooth(phase, width=0.5) 
            
        # 4. Compute Dynamic LFO Cutoff Sweeps
        lfo = (np.sin(2.0 * np.pi * lfo_speed * t) + 1.0) / 2.0

        max_cutoff = random.uniform(2500.0, 4500.0) # Randomizes the screaming peak ceiling
        cutoff_frequencies = min_cutoff + (lfo * (max_cutoff - min_cutoff))

            high_state = raw_oscillator[i] - low_state - q1 * band_state

            if driven_block[i] > pos_clip:     driven_block[i] = pos_clip
            elif driven_block[i] < neg_clip:   driven_block[i] = neg_clip
            
        master_audio_array = np.concatenate((master_audio_array, driven_block))

The Verdict on Pacing: 2 Seconds vs. The Punk Beat
For a standard video, 2 seconds per image is a good rule of thumb. However, for a high-intensity, raw anti-war track like WAR, a flat 2-second delay might feel a bit too rigid or slow.
Instead of a fixed timer, the absolute best way to pace an anarcho-punk slideshow is to match the frames to the tempo (BPM) of the music.
The 1-Second Cut (On the Beat): If the song is fast (around 120 BPM), changing the image every 1.0 seconds aligns perfectly with the heavy thud of the snare drum. This creates a relentless, rhythmic pulse.
The 2-Second Cut (On the Half-Beat): Changing every 2.0 seconds hits exactly on every second bar of music. This gives the viewer just enough time to process the image before the next one slams into place.
The 4-Frame Flash (The Chaos Cut): For intense vocal peaks from Rags or a heavy guitar explosion from you and Maggie, flashing an image for just 0.1 to 0.2 seconds (4 to 5 frames) creates a jarring, strobe-like effect that feels incredibly urgent.

import os
from moviepy.editor import ImageClip, concatenate_videoclips

def create_slideshow(folder_path, output_name="slideshow_output.mp4", duration_per_image=1.5):
    print(f"Scanning folder '{folder_path}' for historical visuals...")
    
    # 1. Gather and sort all images in the folder
    supported_exts = ('.png', '.jpg', '.jpeg', '.bmp')
    image_files = [
        os.path.join(folder_path, f) for f in os.listdir(folder_path) 
        if f.lower().endswith(supported_exts)
    ]
    
    if not image_files:
        print("Error: No valid images found in that folder!")
        return
        
    image_files.sort()  # Keeps them in numeric/alphabetical order
    print(f"Found {len(image_files)} files. Building timeline at {duration_per_image}s per cut...")
    
    clips = []
    for img_path in image_files:
        try:
            # 2. Convert image to a video clip with a fixed duration
            # We set a standard YouTube widescreen resolution (1920x1080)
            clip = ImageClip(img_path).set_duration(duration_per_image)

import os
import requests

def download_from_list(text_file, target_folder):
    # 1. Create the output folder if it doesn't exist
    os.makedirs(target_folder, exist_ok=True)
    
    if not os.path.exists(text_file):
        print(f"Error: Can't find '{text_file}'. Make sure it's in this folder!")
        return

    # 2. Read the links from your text file
    with open(text_file, 'r') as f:
        links = [line.strip() for line in f if line.strip()]

    print(f"Found {len(links)} image URLs in your list. Starting download...")
    
    count = 1
    for url in links:
        try:
            # 3. Stream the file data safely
            response = requests.get(url, timeout=10)
            if response.status_code == 200:
                # Figure out the file extension from the URL (default to .jpg)
                ext = url.split('.')[-1].split('?')[0].lower()
                if ext not in ['jpg', 'jpeg', 'png', 'gif']:
                    ext = 'jpg'
                
                filename = f"{target_folder}/{count:06d}.{ext}"
                
                # 4. Save the image to your folder
                with open(filename, 'wb') as handler:
                    handler.write(response.content)
                
                print(f"Saved: {filename}")
                count += 1
            else:
                print(f"Failed to fetch (Status {response.status_code}): {url}")
        except Exception as e:
            print(f"Error downloading {url}: {e}")

    print(f"\nDone! All images saved cleanly into the '{target_folder}' folder.")

def convert_and_clean_folder(folder_path):
    print("Checking for modern web formats (.webp / .avif)...")
    supported_exts = ('.png', '.jpg', '.jpeg', '.bmp')
    modern_exts = ('.webp', '.avif')
    
    for f in os.listdir(folder_path):
        if f.lower().endswith(modern_exts):
            full_path = os.path.join(folder_path, f)
            try:
                # Open the webp/avif image and convert to RGB (standard JPG)
                with Image.open(full_path) as img:
                    rgb_img = img.convert('RGB')
                    # Save it as a clean new .jpg file
                    new_filename = os.path.splitext(full_path)[0] + ".jpg"
                    rgb_img.save(new_filename, 'JPEG')
                # Delete the old stubborn webp/avif file so it doesn't clutter the folder
                os.remove(full_path)
                print(f"Successfully converted modern format to: {new_filename}")
            except Exception as e:
                print(f"Could not convert file {f}: {e}")
