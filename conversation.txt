def studio_crossover_filter(audio_array, cutoff_freq, filter_type='low', sr=48000):
    from scipy.signal import butter, sosfiltfilt
    nyquist = 0.5 * sr
    if isinstance(cutoff_freq, (list, tuple)):
        low_cut = np.clip(cutoff_freq[0], 20.0, nyquist - 100.0)
        high_cut = np.clip(cutoff_freq[1], 20.0, nyquist - 100.0)
        normal_cutoff = [low_cut / nyquist, high_cut / nyquist]
    else:
        cutoff_freq = np.clip(cutoff_freq, 20.0, nyquist - 100.0)
        normal_cutoff = cutoff_freq / nyquist
    btype = 'lowpass' if filter_type == 'low' else ('highpass' if filter_type == 'high' else 'bandpass')
    sos = butter(4, normal_cutoff, btype=btype, output='sos')
    return sosfiltfilt(sos, audio_array, axis=0)

    out_shred   = cv2.VideoWriter("stem4_shred.avi", fourcc, fps, (w, h))

    # Initialize a clean canvas buffer for the recursive feedback loop
    feedback_canvas = img.copy().astype(np.float32)

    print(f" -> Spinning 4 distinct algorithmic timelines across {total_frames} frames...")

        t = f_idx / total_frames
        osc_fast = (np.sin(t * np.pi * 8) + 1.0) / 2.0

        f_melt = img.copy()

            f_melt[r, :] = np.roll(f_melt[r, :], offset, axis=0)
        # Apply interlaced lines
        f_melt[0:h:2, :, :] = (f_melt[0:h:2, :, :] * 0.85).astype(np.uint8)
        out_melt.write(f_melt)

        f_roll = img.copy()

        if osc_fast > 0.7:

    if not os.path.exists(video_path):
        print(f"❌ Error: Missing source video file '{video_path}'")
        return

    # 1. Open the video file stream
    cap = cv2.VideoCapture(video_path)
    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    fps = cap.get(cv2.CAP_PROP_FPS)
    
    if total_frames <= 0:
        print("❌ Error: Could not read frames from video file.")
        return

    duration = total_frames / fps
    print(f" -> Source Asset: {video_path} ({duration:.2f} seconds, {total_frames} total frames)")
    print(f" -> Extraction target: Splitting timeline into {num_stills} unique stills...")

    # Extract base name of the video file for automated naming
    base_name = os.path.splitext(os.path.basename(video_path))[0]

    # 2. Calculate the exact frame step interval to spread the stills evenly
    step = max(1, total_frames // num_stills)
    extracted_count = 0

    # 3. Core Extraction Loop
    for i in range(num_stills):
        target_frame = i * step
        if target_frame >= total_frames:
            break
            
        # Force the video reader position to jump directly to our target frame coordinate
        cap.set(cv2.CAP_PROP_POS_FRAMES, target_frame)
        success, frame = cap.read()
        
        if success:
            # Construct a clean, automated file name using our INFECTED system prefix
            output_filename = f"INFECTED_EXTRACT_{base_name}_frame_{i+1:02d}.jpg"

            cv2.imwrite(output_filename, frame, [int(cv2.IMWRITE_JPEG_QUALITY), 95])
            print(f"    ✔️ Quarried still saved: {output_filename}")
            extracted_count += 1
        else:
            print(f"    ⚠️ Failed to extract frame at index {target_frame}")

    cap.release()
